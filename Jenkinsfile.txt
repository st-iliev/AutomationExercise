pipeline {
    agent any
    stages {
        stage('Restore') {
            steps {
                bat 'dotnet restore "Exercises Selenium Advanced and POM.sln"'
            }
        }
        stage('Build') {
            steps {
                bat 'dotnet build "Exercises Selenium Advanced and POM.sln" --configuration Release'
            }
        }
        stage('Test') {
            steps {
                bat 'dotnet test "Exercises Selenium Advanced and POM.sln"'
            }
        }
    }
}
