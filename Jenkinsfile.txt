pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                bat 'dotnet restore'
                bat 'dotnet build --configuration Release'
            }
        }

        stage('Run Tests') {
            steps {
                // Start Chrome WebDriver
                bat 'start /B chromedriver --port=9515 --url-base=/wd/hub'
                // Run tests
                bat 'dotnet test "Automation Exercise/AutomationExercise.csproj" --configuration Release'
            }
        }

        stage('Restore and Build Leopard-Report-API') {
            steps {
                // Restore, build, and test Leopard-Report-API project
                bat 'dotnet restore "Automation Exercise.sln"'
                bat 'dotnet build "Automation Exercise.sln" --configuration Release'
                bat 'dotnet test "Automation Exercise/AutomationExercise.csproj"'
            }
        }
    }

    post {
        always {
            bat 'taskkill /F /IM chromedriver.exe /T || true'
        }
    }
}
